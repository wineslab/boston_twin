355    def add_scene_antennas(
356        self,
357        tx_antenna_ids: Union[List[int], np.typing.ArrayLike],
358        rx_antenna_ids: Union[List[int], np.typing.ArrayLike],
359        tx_names: List[str] = [],
360        rx_names: List[str] = [],
361        tx_params: list = [],
362        rx_params: list = [],
363    ) -> dict:
364        """Add antennas to the current scene.
365
366        Parameters
367        ----------
368        tx_antenna_ids : List[int]
369            List of indices of the antennas to be used as Transmitters.
370        rx_antenna_ids : List[int]
371            List of IDs to be used as Receiver names (RX_{id}).
372        tx_names : List[str], optional
373            List of Transmitter names (Default: TX_{id}). Defaults to [].
374        rx_names : List[str], optional
375            List of Receiver names (Default: RX_{id}). Defaults to [].
376        tx_params : list, optional
377            Parameters for the Transmitters. Refer to the Sionna Ray Tracer documentation. Defaults to [].
378        rx_params : list, optional
379            Parameters for the Receivers. Refer to the Sionna Ray Tracer documentation. Defaults to [].
380
381        Returns
382        -------
383        nodes_dict : dict
384            Dictionary with the names of the Transmitters/Receivers as keys and the corresponding Sionna object as values.
385        """
386        txrx_ids = np.concatenate([tx_antenna_ids, rx_antenna_ids])
387        self._current_scene_txrx_localcrs = self._current_scene_antennas_localcrs.loc[
388            txrx_ids, :
389        ]
390        self._current_scene_txrx_localcrs.loc[tx_antenna_ids, "TX/RX"] = "TX"
391        self._current_scene_txrx_localcrs.loc[rx_antenna_ids, "TX/RX"] = "RX"
392
393        if not tx_names:
394            tx_names = [f"TX_{i}" for i in range(len(tx_antenna_ids))]
395        self._current_scene_txrx_localcrs.loc[tx_antenna_ids, "Name"] = tx_names
396
397        for tx_idx, (tx_name, tx_antenna_idx) in enumerate(
398            zip(tx_names, tx_antenna_ids)
399        ):
400            antenna_coords = list(
401                self._current_scene_antennas_localcrs.loc[
402                    tx_antenna_idx, "geometry"
403                ].coords[0]
404            )
405            antenna_coords.append(self.node_height)
406            if len(tx_params) > 0:
407                tx_par = tx_params[tx_idx]
408            else:
409                tx_par = {}
410            tx = Transmitter(tx_name, position=antenna_coords, **tx_par)
411            self.current_sionna_scene.add(tx)
412            self.txs.append(tx)
413
414        if not rx_names:
415            rx_names = [f"RX_{i}" for i in range(len(rx_antenna_ids))]
416        self._current_scene_txrx_localcrs.loc[rx_antenna_ids, "Name"] = rx_names
417
418        for rx_idx, (rx_name, rx_antenna_idx) in enumerate(
419            zip(rx_names, rx_antenna_ids)
420        ):
421            antenna_coords = list(
422                self._current_scene_antennas_localcrs.loc[
423                    rx_antenna_idx, "geometry"
424                ].coords[0]
425            )
426            antenna_coords.append(self.node_height)
427            if len(rx_params) > 0:
428                rx_par = rx_params[rx_idx]
429            else:
430                rx_par = {}
431            rx = Receiver(rx_name, position=antenna_coords, **rx_par)
432            self.current_sionna_scene.add(rx)
433            self.rxs.append(rx)
434
435        nodes_dict = dict(zip(tx_names + rx_names, self.txs + self.rxs))
436        return nodes_dict

438    def generate_scene_from_radius(
439        self,
440        scene_name: str,
441        center_lon: float,
442        center_lat: float,
443        side_m: float,
444        load=False,
445    ):
446        """Generate a new scene specifying its center and radius.
447
448        Parameters
449        ----------
450        scene_name : str
451            Name of the new scene.
452        center_lon : float
453            Longitude of the center.
454        center_lat : float
455            Latitude of the center.
456        side_m : float
457            Radius of the scene.
458        load : bool, optional
459            Load the scene as current scene. Defaults to False.
460        """
461        radius = np.sqrt(2) * side_m  # m
462        azimuths = [45, 225]
463
464        geod = pyproj.Geod(ellps="WGS84")
465        lon1, lat1, _ = geod.fwd(center_lon, center_lat, azimuths[0], radius)
466        lon2, lat2, _ = geod.fwd(center_lon, center_lat, azimuths[1], radius)
467        bbox = [lon1, lat1, lon2, lat2]
468        print("Selecting models within the area...")
469        t0 = time.time()
470        boston_gdf = gpd.GeoDataFrame.from_file(
471            self.boston_model_path.joinpath("boston.geojson"), bbox=bbox
472        )
473        t1 = time.time()
474        print(f"Done. ({t1-t0:.2f} s)")
475        self.boston_model.generate_scene_from_model_gdf(
476            boston_gdf, (center_lon, center_lat), scene_name
477        )
478
479        if load:
480            self.set_scene(scene_name)
