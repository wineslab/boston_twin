350    def add_scene_antennas(
351        self,
352        tx_antenna_ids: Union[List[int], np.typing.ArrayLike],
353        rx_antenna_ids: Union[List[int], np.typing.ArrayLike],
354        tx_names: List[str] = [],
355        rx_names: List[str] = [],
356        tx_params: list = [],
357        rx_params: list = [],
358    ) -> dict:
359        """Add antennas to the current scene.
360
361        Parameters
362        ----------
363        tx_antenna_ids : List[int]
364            List of indices of the antennas to be used as Transmitters.
365        rx_antenna_ids : List[int]
366            List of IDs to be used as Receiver names (RX_{id}).
367        tx_names : List[str], optional
368            List of Transmitter names (Default: TX_{id}). Defaults to [].
369        rx_names : List[str], optional
370            List of Receiver names (Default: RX_{id}). Defaults to [].
371        tx_params : list, optional
372            Parameters for the Transmitters. Refer to the Sionna Ray Tracer documentation. Defaults to [].
373        rx_params : list, optional
374            Parameters for the Receivers. Refer to the Sionna Ray Tracer documentation. Defaults to [].
375
376        Returns
377        -------
378        nodes_dict : dict
379            Dictionary with the names of the Transmitters/Receivers as keys and the corresponding Sionna object as values.
380        """
381        txrx_ids = np.concatenate([tx_antenna_ids, rx_antenna_ids])
382        self.current_scene_txrx_localcrs = self.current_scene_antennas_localcrs.loc[
383            txrx_ids, :
384        ]
385        self.current_scene_txrx_localcrs.loc[tx_antenna_ids, "TX/RX"] = "TX"
386        self.current_scene_txrx_localcrs.loc[rx_antenna_ids, "TX/RX"] = "RX"
387
388        if not tx_names:
389            tx_names = [f"TX_{i}" for i in range(len(tx_antenna_ids))]
390        self.current_scene_txrx_localcrs.loc[tx_antenna_ids, "Name"] = tx_names
391
392        for tx_idx, (tx_name, tx_antenna_idx) in enumerate(
393            zip(tx_names, tx_antenna_ids)
394        ):
395            antenna_coords = list(
396                self.current_scene_antennas_localcrs.loc[
397                    tx_antenna_idx, "geometry"
398                ].coords[0]
399            )
400            antenna_coords.append(self.node_height)
401            if len(tx_params) > 0:
402                tx_par = tx_params[tx_idx]
403            else:
404                tx_par = {}
405            tx = Transmitter(tx_name, position=antenna_coords, **tx_par)
406            self.current_sionna_scene.add(tx)
407            self.txs.append(tx)
408
409        if not rx_names:
410            rx_names = [f"RX_{i}" for i in range(len(rx_antenna_ids))]
411        self.current_scene_txrx_localcrs.loc[rx_antenna_ids, "Name"] = rx_names
412
413        for rx_idx, (rx_name, rx_antenna_idx) in enumerate(
414            zip(rx_names, rx_antenna_ids)
415        ):
416            antenna_coords = list(
417                self.current_scene_antennas_localcrs.loc[
418                    rx_antenna_idx, "geometry"
419                ].coords[0]
420            )
421            antenna_coords.append(self.node_height)
422            if len(rx_params) > 0:
423                rx_par = rx_params[rx_idx]
424            else:
425                rx_par = {}
426            rx = Receiver(rx_name, position=antenna_coords, **rx_par)
427            self.current_sionna_scene.add(rx)
428            self.rxs.append(rx)
429
430        nodes_dict = dict(zip(tx_names + rx_names, self.txs + self.rxs))
431        return nodes_dict

433    def generate_scene_from_radius(
434        self,
435        scene_name: str,
436        center_lon: float,
437        center_lat: float,
438        side_m: float,
439        load=False,
440    ):
441        """Generate a new scene specifying its center and radius.
442
443        Parameters
444        ----------
445        scene_name : str
446            Name of the new scene.
447        center_lon : float
448            Longitude of the center.
449        center_lat : float
450            Latitude of the center.
451        side_m : float
452            Radius of the scene.
453        load : bool, optional
454            Load the scene as current scene. Defaults to False.
455        """
456        radius = np.sqrt(2) * side_m  # m
457        azimuths = [45, 225]
458
459        geod = pyproj.Geod(ellps="WGS84")
460        lon1, lat1, _ = geod.fwd(center_lon, center_lat, azimuths[0], radius)
461        lon2, lat2, _ = geod.fwd(center_lon, center_lat, azimuths[1], radius)
462        bbox = [lon1, lat1, lon2, lat2]
463        print("Selecting models within the area...")
464        t0 = time.time()
465        boston_gdf = gpd.GeoDataFrame.from_file(
466            self.boston_model_path.joinpath("boston.geojson"), bbox=bbox
467        )
468        t1 = time.time()
469        print(f"Done. ({t1-t0:.2f} s)")
470        self.boston_model.generate_scene_from_model_gdf(
471            boston_gdf, (center_lon, center_lat), scene_name
472        )
473
474        if load:
475            self.set_scene(scene_name)

477    def export_scene_antennas(self, out_path: Union[Path, str]):
478        """Export to file the location (in the local CRS) of the antennas in the current scene.
479
480        Parameters
481        ----------
482        out_path : Union[Path,str]
483            Path where to export the antenna location.
484        """
485        if not isinstance(out_path, Path):
486            out_path = Path(out_path)
487
488        if out_path.suffix.lower() != ".json":
489            raise ValueError("Currently, only JSON is supported as export format.")
490
491        if out_path.is_file():
492            raise FileExistsError(f"File {out_path} already exists!")
493
494        if not out_path.parent.is_dir():
495            raise FileNotFoundError(f"Directory {out_path.parent} not found!")
496
497        with open(out_path, "w") as f:
498            json.dump(self._node_pos_dict, f, indent=4)

500    def export_scene_models(self, out_path: Union[Path,str]):
501        """Copy the scene files (`<scene_name>.xml`, `<scene_name>.geojson`, `<scene_name>_tileinfo.geojson`, and the corresponding PLY meshes) to `out_path`.
502        
503        Parameters
504        ----------
505        out_path : Union[Path,str]
506            Path where to export the antenna location.
507        """
508        if not isinstance(out_path, Path):
509            out_path = Path(out_path)
510
511        if out_path.suffix.lower() != "":
512            raise ValueError(f"out_path must point to a folder. Instead {out_path}")
513
514        if not out_path.is_dir():
515            out_path.mkdir(parents=True)
516
517        # scene XML
518        new_mi_scene_path = out_path.with_name(self.mi_scene_path.name)
519        shutil.copy(self.mi_scene_path, new_mi_scene_path)
520
521        # scene GeoJSON
522        new_geo_scene_path = out_path.with_name(self.geo_scene_path.name)
523        shutil.copy(self.geo_scene_path, new_geo_scene_path)
524
525        # scene tileinfo
526        new_tile_info_path = out_path.with_name(self.tile_info_path.name)
527        shutil.copy(self.tile_info_path, new_tile_info_path)
528
529        # copy meshes
530        new_mesh_path = out_path.joinpath("meshes")
531        new_mesh_path.mkdir(exist_ok=True, parents=True)
532        for mesh_id in self.boston_model.tiles_dict[self.scene_name]["models"]:
533            mesh_in_path = self.boston_model.mesh_dir.joinpath(mesh_id + ".ply")
534            mesh_out_path = self.new_mesh_path.joinpath(mesh_id + ".ply")
535            shutil.copy(mesh_in_path,mesh_out_path)

351    def add_scene_antennas(
352        self,
353        tx_antenna_ids: Union[List[int], np.typing.ArrayLike],
354        rx_antenna_ids: Union[List[int], np.typing.ArrayLike],
355        tx_names: List[str] = [],
356        rx_names: List[str] = [],
357        tx_params: list = [],
358        rx_params: list = [],
359    ) -> dict:
360        """Add antennas to the current scene.
361
362        Parameters
363        ----------
364        tx_antenna_ids : List[int]
365            List of indices of the antennas to be used as Transmitters.
366        rx_antenna_ids : List[int]
367            List of IDs to be used as Receiver names (RX_{id}).
368        tx_names : List[str], optional
369            List of Transmitter names (Default: TX_{id}). Defaults to [].
370        rx_names : List[str], optional
371            List of Receiver names (Default: RX_{id}). Defaults to [].
372        tx_params : list, optional
373            Parameters for the Transmitters. Refer to the Sionna Ray Tracer documentation. Defaults to [].
374        rx_params : list, optional
375            Parameters for the Receivers. Refer to the Sionna Ray Tracer documentation. Defaults to [].
376
377        Returns
378        -------
379        nodes_dict : dict
380            Dictionary with the names of the Transmitters/Receivers as keys and the corresponding Sionna object as values.
381        """
382        txrx_ids = np.concatenate([tx_antenna_ids, rx_antenna_ids])
383        self.current_scene_txrx_localcrs = self.current_scene_antennas_localcrs.loc[
384            txrx_ids, :
385        ]
386        self.current_scene_txrx_localcrs.loc[tx_antenna_ids, "TX/RX"] = "TX"
387        self.current_scene_txrx_localcrs.loc[rx_antenna_ids, "TX/RX"] = "RX"
388
389        if not tx_names:
390            tx_names = [f"TX_{i}" for i in range(len(tx_antenna_ids))]
391        self.current_scene_txrx_localcrs.loc[tx_antenna_ids, "Name"] = tx_names
392
393        for tx_idx, (tx_name, tx_antenna_idx) in enumerate(
394            zip(tx_names, tx_antenna_ids)
395        ):
396            antenna_coords = list(
397                self.current_scene_antennas_localcrs.loc[
398                    tx_antenna_idx, "geometry"
399                ].coords[0]
400            )
401            antenna_coords.append(self.node_height)
402            if len(tx_params) > 0:
403                tx_par = tx_params[tx_idx]
404            else:
405                tx_par = {}
406            tx = Transmitter(tx_name, position=antenna_coords, **tx_par)
407            self.current_sionna_scene.add(tx)
408            self.txs.append(tx)
409
410        if not rx_names:
411            rx_names = [f"RX_{i}" for i in range(len(rx_antenna_ids))]
412        self.current_scene_txrx_localcrs.loc[rx_antenna_ids, "Name"] = rx_names
413
414        for rx_idx, (rx_name, rx_antenna_idx) in enumerate(
415            zip(rx_names, rx_antenna_ids)
416        ):
417            antenna_coords = list(
418                self.current_scene_antennas_localcrs.loc[
419                    rx_antenna_idx, "geometry"
420                ].coords[0]
421            )
422            antenna_coords.append(self.node_height)
423            if len(rx_params) > 0:
424                rx_par = rx_params[rx_idx]
425            else:
426                rx_par = {}
427            rx = Receiver(rx_name, position=antenna_coords, **rx_par)
428            self.current_sionna_scene.add(rx)
429            self.rxs.append(rx)
430
431        nodes_dict = dict(zip(tx_names + rx_names, self.txs + self.rxs))
432        return nodes_dict

435        self,
436        scene_name: str,
437        center_lon: float,
438        center_lat: float,
439        side_m: float,
440        load=False,
441    ):
442        """Generate a new scene specifying its center and radius.

444        Parameters

463        bbox = [lon1, lat1, lon2, lat2]
464        print("Selecting models within the area...")
465        t0 = time.time()
466        boston_gdf = gpd.GeoDataFrame.from_file(
467            self.boston_model_path.joinpath("boston.geojson"), bbox=bbox
468        )
469        t1 = time.time()
470        print(f"Done. ({t1-t0:.2f} s)")
471        self.boston_model.generate_scene_from_model_gdf(
472            boston_gdf, (center_lon, center_lat), scene_name
473        )
474
475        if load:
476            self.set_scene(scene_name)
